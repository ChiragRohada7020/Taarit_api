How to make an API call

Post Data

Driver Registration

                
                  syntax ~$  https://taarit-app1.onrender.com/ambulance_register/{api}
                
              
                
                  Parametrs ~$  Name , Email , Password , Address , Mobile
                
              
                
                  const data = {
                    name: "Your_Name",
                    mobile: "Mobile_Number",
                    address: "Your_Address",
                    email: "Your_Email",
                    password: "Your_Password",
                  };

fetch("/ambulance_register", {
  method: "POST", // or 'PUT'
  headers: {
    "Content-Type": "application/json",
  },
  body: JSON.stringify(data),
})
  .then((response) => response.json())
  .then((data) => {
    console.log("Success:", data);
  })
  .catch((error) => {
    console.error("Error:", error);
  });
